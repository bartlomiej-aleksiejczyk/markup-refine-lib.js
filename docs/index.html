Demonstration of markup-refine-lib.js Library

This page contains a demonstration of how to use the markup-refine-lib.js web components.

Dynamic Topbar (Presented Above)

To use the dynamic topbar component, add this custom element to your markup: 

        <dynamic-topbar></dynamic-topbar>

Then attach the .data property with topbar content via JavaScript, as shown in the code below: 

        
<script type="module">
const sampleDate = {
  topbarSettings: {
    title: "Dynamic topbar demonstration",
    logoUri:
      "https://img.icons8.com/?size=100&id=baTWeZAqG8lF&format=png&color=ffffff",
    homeUri: "/markup-refine-lib.js/",
  },
  fixedContent: {
    logoutUrl: "https://mysite.invalid/logout",
    userPreferencesUrl: "https://mysite.invalid/pref",
    userSettingsUrl: "https://mysite.invalid/sett",
  },
  content: [
    {
      link: "https://mysite.invalid/users/",
      iconName: "Customers",
      defaultName: "Customers",
      id: "users",
      type: "link",
    },
    {
      defaultName: "Custom Settings",
      type: "container",
      id: "settings",
      objects: [
        {
          link: "https://mysite.invalid/security/",
          iconName: "shield",
          defaultName: "Security",
          id: "security",
          type: "link",
        },
        {
          link: "https://mysite.invalid/privacy/",
          defaultName: "Privacy",
          id: "privacy",
          type: "link",
        },
        {
          link: "https://mysite.invalid/privacy/",
          defaultName: "GDPR",
          id: "privacy",
          type: "link",
        },
      ],
    },
  ],
};
document.addEventListener("DOMContentLoaded", () => {
  const topBar = document.querySelector("dynamic-topbar");
  if (topBar) {
    topBar.data = JSON.stringify(sampleDate);
  }
});
</script>
